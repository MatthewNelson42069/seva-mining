```python
async def _check_compliance(self, draft_text: str) -> bool:
    """Haiku compliance check. Returns True if draft passes, False if blocked."""
    response = await self.anthropic.messages.create(
        model="claude-haiku-3-20240307",
        max_tokens=200,
        system="You are a compliance checker...",
        messages=[{"role": "user", "content": f"Check this draft: {draft_text}"}],
    )
    result_text = response.content[0].text.strip().lower()
    # Fail-safe: ambiguous response = block
    if "pass" in result_text:
        return True
    return False
```
