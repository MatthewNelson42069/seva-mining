    ```tsx
    import { format } from 'date-fns'
    import { ChevronLeft, ChevronRight } from 'lucide-react'
    import { Button } from '@/components/ui/button'
    import { getWeekEnd, getWeekStart } from '@/lib/week'

    interface WeekNavProps {
      weekAnchor: Date
      onPrev: () => void
      onNext: () => void
      onToday: () => void
    }

    /**
     * v2.1 Phase 6 — Week navigation header (CAL-05 / D-08).
     *
     * Layout: [prev] [Today] [next]   <label>
     * The component is stateless w/r/t which week is showing; the parent
     * (ContentCalendarPage) owns `weekAnchor` state and updates it via the
     * three callbacks.
     */
    export function WeekNav({ weekAnchor, onPrev, onNext, onToday }: WeekNavProps) {
      const start = getWeekStart(weekAnchor)
      const end = getWeekEnd(weekAnchor)
      // Label like "May 18 – May 24, 2026". If the week spans a year boundary,
      // both halves include the year.
      const sameYear = start.getFullYear() === end.getFullYear()
      const startLabel = sameYear ? format(start, 'MMM d') : format(start, 'MMM d, yyyy')
      const endLabel = format(end, 'MMM d, yyyy')

      return (
        <div className="flex items-center justify-between gap-3 mb-4">
          <div className="flex items-center gap-1.5">
            <Button
              variant="outline"
              size="sm"
              onClick={onPrev}
              aria-label="Previous week"
            >
              <ChevronLeft className="h-4 w-4" />
            </Button>
            <Button
              variant="outline"
              size="sm"
              onClick={onToday}
              aria-label="Jump to today"
            >
              Today
            </Button>
            <Button
              variant="outline"
              size="sm"
              onClick={onNext}
              aria-label="Next week"
            >
              <ChevronRight className="h-4 w-4" />
            </Button>
          </div>
          <span className="text-sm text-zinc-400" data-testid="week-range-label">
            {startLabel} – {endLabel}
          </span>
        </div>
      )
    }
    ```

    ```tsx
    import { useCallback, useState } from 'react'
    import { addDays } from 'date-fns'
    import { WeekNav } from '@/components/calendar/WeekNav'
    import { WeeklyGrid } from '@/components/calendar/WeeklyGrid'

    /**
     * v2.1 Phase 6 — Content Calendar page (CAL-05).
     *
     * State: `weekAnchor` (a Date inside the visible week). Defaults to today.
     * Layout (inside TabbedDashboard's outer chrome from Phase 5):
     *   [WeekNav: prev/Today/next  |  May 18 – May 24, 2026]
     *   [WeeklyGrid: 7 day cells]
     *
     * The page intentionally has NO tab strip / header — TabbedDashboard
     * already renders TabNav above the <Outlet />.
     */
    export default function ContentCalendarPage() {
      const [weekAnchor, setWeekAnchor] = useState<Date>(() => new Date())

      const handlePrev = useCallback(() => {
        setWeekAnchor((d) => addDays(d, -7))
      }, [])

      const handleNext = useCallback(() => {
        setWeekAnchor((d) => addDays(d, 7))
      }, [])

      const handleToday = useCallback(() => {
        setWeekAnchor(new Date())
      }, [])

      return (
        <div className="px-4 py-6">
          <WeekNav
            weekAnchor={weekAnchor}
            onPrev={handlePrev}
            onNext={handleNext}
            onToday={handleToday}
          />
          <WeeklyGrid weekAnchor={weekAnchor} />
        </div>
      )
    }
    ```

    Keep the `default export` style — `frontend/src/App.tsx` (from Phase 5 Plan 05) already imports `ContentCalendarPage` as a default import. NO App.tsx changes needed.
