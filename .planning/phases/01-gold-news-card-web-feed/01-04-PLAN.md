    """GET /summaries — v2.0 daily summary feed read endpoint.

    Phase 1, Plan 04. Auth-gated at router level (mirrors digests.py — single-user
    JWT contract). Returns up to 120 rows ordered by generated_at DESC.

    Requirement: FEED-05 (auth-gated read).
    """
    from fastapi import APIRouter, Depends, Query
    from sqlalchemy import select
    from sqlalchemy.ext.asyncio import AsyncSession

    from app.database import get_db
    from app.dependencies import get_current_user
    from app.models.daily_summary import DailySummary
    from app.schemas.daily_summary import SummaryCardResponse, SummaryFeedResponse

    router = APIRouter(
        prefix="/summaries",
        tags=["summaries"],
        dependencies=[Depends(get_current_user)],  # FEED-05 — router-level auth
    )


    @router.get("", response_model=SummaryFeedResponse)
    async def list_summaries(
        limit: int = Query(60, ge=1, le=120),
        db: AsyncSession = Depends(get_db),
    ) -> SummaryFeedResponse:
        """Return up to `limit` summaries ordered by generated_at DESC.

        30-day retention × 2 fires/day = 60 rows max in steady state. The 120
        ceiling exists for forensic convenience without exposing an unbounded
        list.
        """
        stmt = (
            select(DailySummary)
            .order_by(DailySummary.generated_at.desc())
            .limit(limit)
        )
        result = await db.execute(stmt)
        rows = result.scalars().all()
        cards = [SummaryCardResponse.model_validate(r) for r in rows]
        return SummaryFeedResponse(summaries=cards, total=len(cards))
    ```

    """Integration tests for GET /summaries — Phase 1, Plan 04.

    Covers FEED-05 (auth-gated), pagination clamps, DESC ordering, response shape.
    """
    from datetime import datetime, timedelta, timezone

    import pytest
    from httpx import AsyncClient

    from app.models.daily_summary import DailySummary


    pytestmark = pytest.mark.asyncio


    async def _insert_summary(db_session, *, generated_at, period_label="08:00 PT",
                              gold_news_md="## Gold News\n* test", status="completed"):
        row = DailySummary(
            generated_at=generated_at,
            period_label=period_label,
            gold_news_md=gold_news_md,
            ontario_law_md=None,
            ontario_stats_md=None,
            raw_sources_jsonb={"gold_news": [], "ontario_law": {"hits": [], "last_known_law": None},
                               "ontario_stats": {"snapshot_date": "", "last_known_figure": None,
                                                  "fresh_data": None}},
            status=status,
            error_text=None,
        )
        db_session.add(row)
        await db_session.commit()
        await db_session.refresh(row)
        return row


    async def test_get_summaries_unauthenticated_returns_401(async_client: AsyncClient):
        resp = await async_client.get("/summaries")
        assert resp.status_code == 401


    async def test_get_summaries_empty_returns_200_with_empty_list(
        async_client: AsyncClient, auth_headers: dict
    ):
        resp = await async_client.get("/summaries", headers=auth_headers)
        assert resp.status_code == 200
        body = resp.json()
        assert body == {"summaries": [], "total": 0}


    async def test_get_summaries_returns_rows_in_descending_order(
        async_client: AsyncClient, auth_headers: dict, db_session
    ):
        now = datetime.now(timezone.utc)
        await _insert_summary(db_session, generated_at=now - timedelta(hours=4))
        await _insert_summary(db_session, generated_at=now - timedelta(hours=8))
        await _insert_summary(db_session, generated_at=now)

        resp = await async_client.get("/summaries", headers=auth_headers)
        assert resp.status_code == 200
        body = resp.json()
        assert body["total"] == 3
        # Newest first
        ts = [c["generated_at"] for c in body["summaries"]]
        assert ts == sorted(ts, reverse=True)


    async def test_get_summaries_respects_limit_param(
        async_client: AsyncClient, auth_headers: dict, db_session
    ):
        now = datetime.now(timezone.utc)
        for i in range(3):
            await _insert_summary(db_session, generated_at=now - timedelta(hours=i))
        resp = await async_client.get("/summaries?limit=2", headers=auth_headers)
        assert resp.status_code == 200
        assert len(resp.json()["summaries"]) == 2


    async def test_get_summaries_limit_above_120_rejected(
        async_client: AsyncClient, auth_headers: dict
    ):
        resp = await async_client.get("/summaries?limit=121", headers=auth_headers)
        assert resp.status_code == 422


    async def test_get_summaries_limit_zero_rejected(
        async_client: AsyncClient, auth_headers: dict
    ):
        resp = await async_client.get("/summaries?limit=0", headers=auth_headers)
        assert resp.status_code == 422


    async def test_get_summaries_response_omits_raw_sources_jsonb(
        async_client: AsyncClient, auth_headers: dict, db_session
    ):
        await _insert_summary(db_session, generated_at=datetime.now(timezone.utc))
        resp = await async_client.get("/summaries", headers=auth_headers)
        assert resp.status_code == 200
        card = resp.json()["summaries"][0]
        assert "raw_sources_jsonb" not in card
        # Required keys present:
        for key in ("id", "generated_at", "period_label", "gold_news_md",
                    "ontario_law_md", "ontario_stats_md", "status", "error_text"):
            assert key in card
    ```
