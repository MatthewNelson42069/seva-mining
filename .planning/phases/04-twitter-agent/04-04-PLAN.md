Modify `_make_job` to handle the twitter_agent job specifically:
```python
def _make_job(job_name: str, engine):
    async def job():
        async with engine.connect() as conn:
            if job_name == "twitter_agent":
                agent = TwitterAgent()
                await with_advisory_lock(
                    conn, JOB_LOCK_IDS[job_name], job_name, agent.run
                )
            else:
                await with_advisory_lock(
                    conn, JOB_LOCK_IDS[job_name], job_name,
                    lambda: placeholder_job(job_name),
                )
    return job
```
